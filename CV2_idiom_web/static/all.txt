let bossMode = false
let answered = false
let counting = false
let stableFrames = 0
let lastGesture = null

let gestureSequence = []
let lastGestureBoss = null
const video = document.getElementById("videoFeed")
const gestureText = document.getElementById("gesture")
const cdEl = document.getElementById("countdown")
const resultEl = document.getElementById("result")

window.onload = function(){
    loadQuestion()
    setInterval(updateVideo, 50)   // ⭐ 每50ms更新一次（20FPS）
}

async function updateVideo() {
    const res  = await fetch("/video_frame")
    const data = await res.json()

    video.src = "data:image/jpeg;base64," + data.image

    // ⭐ Boss模式：只處理技能
    if (bossMode) {
        if (data.gesture) {
            detectCombo(data.gesture)
        }
        return
    }

    clearHighlight()

    if (answered) return

    if (data.gesture) {

        gestureText.innerText = "偵測手勢: " + data.gesture
        highlightOption(data.gesture)

        if (data.gesture === lastGesture) {

            stableFrames++

            if (stableFrames > 10 && !counting) {
                startCountdown(data.gesture)
            }

        } else {

            lastGesture  = data.gesture
            stableFrames = 0

            if (counting) {
                counting = false
                cdEl.innerText = ""
            }
        }

    } else {

        gestureText.innerText = ""
        lastGesture  = null
        stableFrames = 0

        if (counting) {
            counting = false
            cdEl.innerText = ""
        }
    }
}


// ── 提交答案 ─────────────────────────
async function submitAnswer(gesture) {

    answered = true

    const res  = await fetch("/check_answer/" + gesture)
    const data = await res.json()

    if (data.correct) {

        resultEl.innerText = "✅ 正確!"
        resultEl.className = "correct"

        setTimeout(() => {
            resultEl.innerText = ""
            startBossBattle()
        }, 1000)

    } else {

        resultEl.innerText = "❌ 錯誤!"
        resultEl.className = "wrong"

        setTimeout(() => {
            resultEl.innerText = ""
            resetGestureState()   // ⭐補這行（避免卡死）
            loadQuestion()
        }, 1500)
    }
}


// ── Boss開始 ─────────────────────────
function startBossBattle(){

    bossMode = true

    // ⭐隱藏畫面
    document.getElementById("quizArea").classList.add("hidden")
    document.getElementById("cameraArea").classList.add("hidden")

    // ⭐顯示Boss
    document.getElementById("bossArea").classList.remove("hidden")

    // ⭐初始化
    document.getElementById("bossImg").src = "/static/boss_idle.png"

    gestureSequence = []
    lastGestureBoss = null

    // ⭐避免上一題狀態殘留
    lastGesture = null
    stableFrames = 0
    counting = false
}


// ── Boss結束 ─────────────────────────
function endBossBattle(){

    bossMode = false

    // ⭐顯示回來
    document.getElementById("quizArea").classList.remove("hidden")
    document.getElementById("cameraArea").classList.remove("hidden")

    // ⭐隱藏Boss
    document.getElementById("bossArea").classList.add("hidden")

    // ⭐重置狀態（關鍵）
    resetGestureState()

    // ⭐確保圖片回復
    document.getElementById("bossImg").src = "/static/boss_idle.png"

    loadQuestion()
}


// ── 技能 ─────────────────────────
function activateSkill(type){

    const text = document.getElementById("skillText")
    const boss = document.getElementById("bossImg")

    if(type === 1){
        text.innerText = "🔥 火焰斬擊！"
        boss.src = "/static/boss_hit1.gif"
    }

    if(type === 2){
        text.innerText = "⚡ 雷電爆破！"
        boss.src = "/static/boss_hit2.gif"
    }

    // ⭐暫停偵測避免重複觸發
    bossMode = false

    setTimeout(()=>{

        boss.src = "/static/boss_idle.png"
        endBossBattle()

    },2000)
}


// ── Combo偵測 ─────────────────────────
function detectCombo(g){

    if(g === lastGestureBoss) return

    lastGestureBoss = g

    gestureSequence.push(g)

    if(gestureSequence.length > 4){
        gestureSequence.shift()
    }

    const combo = gestureSequence.join("")

    if(combo === "1234"){
        activateSkill(1)
    }

    if(combo === "4321"){
        activateSkill(2)
    }
}

detector = HandDetector()
cap = cv2.VideoCapture(0)
question = new_question()


@app.route("/")
def index():
    return render_template("index.html")


@app.route("/get_question")
def get_question():
    return jsonify(question)


@app.route("/check_answer/<int:gesture>")
def check_answer(gesture):
    global question

    # ✅ 先比對，再換題（原本換題在比對前，導致比對到下一題）
    correct = (gesture == question["correct_index"])
    correct_answer = question["correct_char"]

    result = {
        "correct": correct,
        "correct_answer": correct_answer
    }

    question = new_question()   # 比對完才換題
    return jsonify(result)


@app.route("/video_frame")
def video_frame():
    global cap

    ret, frame = cap.read()
    if not ret:
        return ""

    frame = cv2.flip(frame, 1)
    gesture, frame = detector.detect(frame)

    _, buffer = cv2.imencode('.jpg', frame)
    jpg_as_text = base64.b64encode(buffer).decode('utf-8')

    return jsonify({
        "image": jpg_as_text,
        "gesture": gesture
    })


if __name__ == "__main__":
    app.run(debug=True)

import mediapipe as mp
from mediapipe.tasks import python
from mediapipe.tasks.python import vision
import cv2

class HandDetector:

    def __init__(self):

        base_options = python.BaseOptions(
            model_asset_path=r"C:\python_web\CV2_idiom_web\b_H\hand_landmarker.task"
        )

        options = vision.HandLandmarkerOptions(
            base_options=base_options,
            num_hands=1,
            running_mode=vision.RunningMode.VIDEO
        )

        self.landmarker = vision.HandLandmarker.create_from_options(options)
        self.timestamp = 0

    def classify_gesture(self, landmarks, hand_label):

        tips = [8,12,16,20]
        pip = [6,10,14,18]

        count = 0

        for t,p in zip(tips,pip):
            if landmarks[t].y < landmarks[p].y:
                count += 1

        if hand_label == "Right":
            if landmarks[4].x > landmarks[3].x:
                count += 1
        else:
            if landmarks[4].x < landmarks[3].x:
                count += 1

        return count

    def detect(self, frame):

        rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)

        mp_image = mp.Image(
            image_format=mp.ImageFormat.SRGB,
            data=rgb
        )

        result = self.landmarker.detect_for_video(mp_image,self.timestamp)
        self.timestamp += 1

        gesture = None

        if result.hand_landmarks:

            hand = result.hand_landmarks[0]
            hand_label = result.handedness[0][0].category_name

            gesture = self.classify_gesture(hand,hand_label)

            for lm in hand:

                x = int(lm.x*frame.shape[1])
                y = int(lm.y*frame.shape[0])

                cv2.circle(frame,(x,y),4,(255,100,0),-1)

        return gesture, frame

import random
from idioms_data import idioms, options_pool

def generate_multiple_choice():

    idiom = random.choice(list(idioms.keys()))
    pos = random.choice(list(idioms[idiom].keys()))

    correct_char = idiom[pos]
    question = idiom[:pos] + "＿" + idiom[pos+1:]

    wrong_opts = []

    for lvl_list in idioms[idiom][pos].values():
        wrong_opts.extend(lvl_list)

    wrong_opts = list(set(wrong_opts))

    while len(wrong_opts) < 3:
        c = random.choice(options_pool)
        if c != correct_char and c not in wrong_opts:
            wrong_opts.append(c)

    wrong_opts = random.sample(wrong_opts,3)

    choices = wrong_opts + [correct_char]
    random.shuffle(choices)

    return {
        "mode":"multiple",
        "idiom":idiom,
        "question":question,
        "choices":choices,
        "correct_char":correct_char,
        "correct_index":choices.index(correct_char)+1   # 修正
    }

def new_question():
    return generate_multiple_choice()
